25  /**
26   * This class decrypts and caches the service IDs and passwords from the
27   * application's protected service ID password file. Although applications can
28   * use this class directly, they will normally use service-id and
29   * service-password properties to initialize services, and those properties in
30   * turn rely on this class.
31   *
32   * As a minimum, the ServiceNow UNID of the application instance must be
33   * provided as the <code>appID</code> property when constructing the service.
34   * This is region-specific so it will normally be stored external to the
35   * application.
36   * <pre>
37   *  {@code
38   *  <bean id="sidCache" type="someCompany.someFramework.serviceid.ServiceIdCache" scope="singleton">
39   *     <property name="appId" value="${app.id}" />
40   *      <!-- The remaining properties are optional, used for test purposes.
41   *      -- These examples show the default values.
42   *      -->
43   *      <property name="masterDir" value="/opt/deploy/protected" />
44   *      <property name="passwordFile" value="credentials.enc" />
45   *      <property name="keyFile" value="private.pem" />
46   *      <property name="keyAlgorithm" value="RSA" />
47   *      <property name="cipherAlgorithm" value="RSA/ECB/PKCS1Padding" />
48   *      <property name="caseSensitive" value="false" type="boolean" />
49   *      <property name="trim" value="true" type="boolean" />
50   *      <property name="delimiter" value="|" type="string" />
51   *  </bean>
52   *  }
53   * </pre>
54   * The remaining properties are optional and serve to override the default
55   * values: <br>
56   * <code>masterDir</code>: The top-level directory where the credentials for
57   * various applications are stored. <br>
58   * <code>passwordFile</code>: The filename of the file containing the service
59   * IDs and encrypted passwords for the application's resources. <br>
60   * <code>keyFile</code>: The filename of the private key file. <br>
61   * <code>keyAlgorithm</code>: The key factory algorithm to use. <br>
62   * <code>cipherAlgorithm</code>: The cipher transformation to use. <br>
63   * <code>caseSensitive</code>: Enable case-sensitive resource lookups. <br>
64   * <code>trim</code>: Trim leading/trailing whitespace from IDs and passwords. <br>
65   * <code>delimiter</code>: Delimiter used to separate ID and password in lookups
66   */

363 	/*
364 	 * Get the service ID and password to use to connect to a resource, possibly
365 	 * after refreshing the cache.
366 	 */
367 	private UserPasswordCredentials retrieveCredentials(String resource)
368 			throws ServiceIdException {
369 		LOG.debug("Looking up service ID credentials for resource {}", resource);
370 		refresh(false);
371 		ServiceIdCredential cred = credentials
372 				.get(adjustResourceName(resource));
373 		if (cred == null) {
374 			throw new ServiceIdException("Resource [" + resource
375 					+ "] is not defined in credentials file");
376 		}
377 		return cred.getUserPasswordCredentials();
378 	}

380 	/*
381 	 * Decrypt a password using the private key.
382 	 */
383 	private String decryptPassword(String encryptedPassword,
384 			PrivateKey privateKey) throws ServiceIdException {
385 		// This code is adapted from Credential.getPassword in the TRS SidUtil
386 		// utility
387 		try {
388 			byte[] decodedPassword = Base64.decodeBase64(encryptedPassword
389 					.getBytes(DEFAULT_CHARSET));
390 			Cipher cipher = Cipher.getInstance(this.cipherAlgorithm);
391 			cipher.init(Cipher.DECRYPT_MODE, privateKey);
392 			byte[] decryptedPassword = cipher.doFinal(decodedPassword);
393 			String pw = new String(decryptedPassword, DEFAULT_CHARSET);
394 			if (this.trim) {
395 				pw = pw.trim();
396 			}
397 			return pw;
398 		} catch (Exception ex) {
399 			throw new ServiceIdException("Failed to decode/decrypt password", ex);
400 		}
401 	}

403 	/*
404 	 * Refresh the cached credentials if they are out of date. If the force flag
405 	 * is true, they are unconditionally reloaded.
406 	 */
407 	private void refresh(boolean force) throws ServiceIdException {
408 		if (!this.didOneTimeInitialization) {
409 			doOneTimeInitialization();
410 		}
411 		if (force || !isUpToDate()) {
412 			synchronized (this) {
413 				if (force || !isUpToDate()) {
414 					LOG.info("Loading service ID credentials from "
415 							+ this.credentialsFile);
416 					long newTimestamp = credentialsFile.lastModified();
417 					PrivateKey newKey = readPrivateKey();
418 
419 					// Credentials and credentialsTimestamp must be
420 					// updated at the same time, in synchronized code,
421 					// so they are mutually consistent.  Note that we
422 					// use the timestamp from before the call to loadCredentials,
423 					// so if the credentials are updated again during the
424 					// load, our credentials will immediately be stale.
425 					this.credentials = loadCredentials(newKey);
426 					this.credentialsTimestamp.set(newTimestamp);
427 				}
428 			}
429 		}
430 	}

432 	/*
433 	 * Initialize the filename. This is based on multiple independent setters so
434 	 * we defer this until we need the credentials.
435 	 */
436 	private synchronized void doOneTimeInitialization() {
437 		if (this.appId == null) {
438 			throw new IllegalStateException(
439 					"ServiceIdCache not initialized properly; need appId");
440 		}
441 		File appDir = new File(this.masterDir, this.appId);
442 		this.credentialsFile = new File(appDir, this.passwordFile);
443 		this.privateKeyFile = new File(appDir, this.keyFile);
444 		this.didOneTimeInitialization = true;
445 	}

447 	/*
448 	 * Checks whether the password file has changed since we last checked it,
449 	 * based on the file's timestamp.
450 	 */
451 	private boolean isUpToDate() {
452 		boolean upToDate = true;
453 		long newTimeStamp = this.credentialsFile.lastModified();
454 		if (newTimeStamp == 0) {
455 			// Credentials file doesn't exist. This can happen in the middle of
456 			// publishing
457 			// new credentials. We will just use the old credentials if we have
458 			// them, otherwise
459 			// it's an error. In any case, it's worth logging.
460 			if (this.credentials == null) {
461 				LOG.error("Unable to access credentials in {}",
462 						this.credentialsFile);
463 				throw new ServiceIdException("Unable to access credentials");
464 			} else {
465 				LOG.info("Using cached credentials because {} disappeared",
466 						this.credentialsFile);
467 			}
468 		} else if (newTimeStamp != 0
469 				&& newTimeStamp != this.credentialsTimestamp.longValue()) {
470 			upToDate = false;
471 		}
472 		return upToDate;
473 	}

475 	/*
476 	 * Read the encrypted service ID password file, parse the entries, and store
477 	 * them in a credentials map.  This does not directly update the state of the
478 	 * ServiceIdCache object, it just returns the new map.  It is the responsibility
479 	 * of the caller to store the map along with the timestamp of the credentials file.
480 	 */
481 	private Map<String, ServiceIdCredential> loadCredentials(
482 			PrivateKey privateKey) throws ServiceIdException {
483 		Map<String, ServiceIdCredential> credentials = new HashMap<>();
484 		BufferedReader creds = null;
485 		try {
486 			creds = new BufferedReader(new InputStreamReader(new FileInputStream(credentialsFile), DEFAULT_CHARSET));
487 			int lineNumber = 0;
488 			String entry;
489 			while ((entry = creds.readLine()) != null) {
490 				lineNumber++;
491 				StringTokenizer st = new StringTokenizer(entry, this.delimiter);
492 				String[] fields = new String[MIN_REQUIRED_FIELDS];
493 				int token = 0;
494 				while (token < MIN_REQUIRED_FIELDS && st.hasMoreTokens()) {
495 					fields[token++] = st.nextToken();
496 				}
497 				if (token < MIN_REQUIRED_FIELDS) {
498 					throw new ServiceIdException(
499 							"Corrupted credentials file [line " + lineNumber
500 									+ " has only " + token + " fields: "
501 									+ credentialsFile);
502 				}
503 				String resource = adjustResourceName(fields[RESOURCE_FIELD]);
504 				String id = fields[SERVICE_ID_FIELD];
505 				if (this.trim) {
506 					id = id.trim();
507 				}
508 				String pw = fields[PASSWORD_FIELD];
509 				credentials.put(resource, new ServiceIdCredential(id, pw,
510 						privateKey));
511 			}
512 			return credentials;
513 		} catch (IOException e) {
514 			throw new ServiceIdException("Can't read credentials file "
515 					+ credentialsFile, e);
516 		} finally {
517 			if (creds != null) {
518 				try {
519 					creds.close();
520 				} catch (Exception e) {
521 					// Ignore error on file close
522 				}
523 			}
524 		}
525 	}

527 	/*
528 	 * Read the private key from this.privateKeyFile.
529 	 */
530 	private PrivateKey readPrivateKey() throws ServiceIdException {
531 		// This code is a slightly modified version of Credential.getPassword
532 		// from the TRS SidUtil utility.
533 		BufferedReader pemReader = null;
534 		try {
535 			pemReader = new BufferedReader(new InputStreamReader(new FileInputStream(this.privateKeyFile), DEFAULT_CHARSET));
536 			String line;
537 			StringBuilder pemKey = new StringBuilder();
538 			boolean foundHdr = false;
539 			boolean foundTrailer = false;
540 			while (!foundTrailer && (line = pemReader.readLine()) != null) {
541 				if (foundHdr) {
542 					if (line.equals(PEM_TRAILER)) {
543 						foundTrailer = true;
544 					} else {
545 						pemKey.append(line);
546 					}
547 				} else if (line.equals(PEM_HEADER)) {
548 					foundHdr = true;
549 				}
550 			}
551 			if (!foundHdr) {
552 				throw new ServiceIdException(
553 						"Missing header in private key file: "
554 								+ this.privateKeyFile);
555 			}
556 			if (!foundTrailer) {
557 				throw new ServiceIdException(
558 						"Missing trailer in private key file: "
559 								+ this.privateKeyFile);
560 			}
561 			byte[] decodedKey = Base64.decodeBase64(pemKey.toString().getBytes(DEFAULT_CHARSET));
562 			PKCS8EncodedKeySpec keySpec = new PKCS8EncodedKeySpec(decodedKey);
563 			KeyFactory keyFactory = KeyFactory.getInstance(this.keyAlgorithm);
564 			return keyFactory.generatePrivate(keySpec);
565 		} catch (FileNotFoundException ex) {
566 			throw new ServiceIdException("Can't open private key file: "
567 					+ this.privateKeyFile);
568 		} catch (IOException ex) {
569 			throw new ServiceIdException("Error reading private key file: "
570 					+ this.privateKeyFile);
571 		} catch (NoSuchAlgorithmException ex) {
572 			throw new ServiceIdException("Can't get " + this.cipherAlgorithm
573 					+ " key factory", ex);
574 		} catch (InvalidKeySpecException ex) {
575 			throw new ServiceIdException("Invalid private key file: "
576 					+ this.privateKeyFile);
577 		} finally {
578 			if (pemReader != null) {
579 				try {
580 					pemReader.close();
581 				} catch (IOException ex) {
582 					// Ignore error closing read-only file
583 				}
584 			}
585 		}
586 	}

588 	/*
589 	 * Convert resources to all lower-case depending on the caseSensitive flag,
590 	 * and trim it based on the trim flag.
591 	 */
592 	private String adjustResourceName(String resource) {
593 		if (this.trim) {
594 			resource = resource.trim();
595 		}
596 		if (!this.caseSensitive) {
597 			// Don't change case of leading service ID, if any
598 			int i = resource.indexOf(this.delimiter);
599 			if (i > 0) {
600 				int tailStart = i + this.delimiter.length();
601 				resource = resource.substring(0, tailStart)
602 						+ resource.substring(tailStart).toLowerCase();
603 			} else {
604 				resource = resource.toLowerCase();
605 			}
606 		}
607 		return resource;
608 	}

610 	// This inner class is used only locally to hold the service ID / password
611 	// combinations in the cache. We do lazy decryption so we hold the encrypted
612 	// password until it is decrypted.
613 	class ServiceIdCredential {
614 		private String serviceId;
615 		private String decryptedPassword; // Set on first call to getPassword()
616 		private String encryptedPassword; // Decrypted on first call to
617 											// getPassword()
618 		private PrivateKey privateKey; // Key to decrypt password
619 
620 		ServiceIdCredential(String serviceId, String encryptedPassword,
621 				PrivateKey privateKey) {
622 			this.serviceId = serviceId;
623 			this.decryptedPassword = null;
624 			this.encryptedPassword = encryptedPassword;
625 			this.privateKey = privateKey;
626 		}
627 
628 		String getServiceId() {
629 			return this.serviceId;
630 		}
631 
632 		String getPassword() throws ServiceIdException {
633 			if (this.decryptedPassword == null) {
634 				this.decryptedPassword = decryptPassword(
635 						this.encryptedPassword, this.privateKey);
636 			}
637 			return this.decryptedPassword;
638 		}
639 
640 		UserPasswordCredentials getUserPasswordCredentials()
641 				throws ServiceIdException {
642 			return new ServiceIdUserPasswordCredentials(getServiceId(),
643 					getPassword());
644 		}
645 	}
